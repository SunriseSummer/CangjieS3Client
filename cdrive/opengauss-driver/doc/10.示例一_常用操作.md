### 示例一: 常用操作

```
import std.database.sql.*
import cangjie_tpc::opengauss.driver.*

//创建数据库连接
func GetConnection(): Connection {
    var url = "opengauss://user:password@localhost:5432/bes?sslmode=disable&search_path=public"
    let drvopt = DriverManager.getDriver("opengauss")
    var drv: Driver
    match (drvopt) {
        case Some(v) => drv = v
        case None =>
            println("cannot get the opengauss driver")
            throw SqlException("cannot get the opengauss driver")
    }
    let ds = drv.open(url)
    return ds.connect()
}

//执行普通SQL语句，创建customer_t1表
func CreateTable(conn: Connection) {
    var sql = "CREATE TABLE customer_t1(c_customer_sk INTEGER, c_customer_name VARCHAR(32))"
    var statement = conn.prepareStatement(sql)
    var updateresult = statement.update()
    statement.close()
}

//执行预处理语句，插入数据
func BatchInsertData(conn: Connection) {
    var sql = "INSERT INTO customer_t1 VALUES (?,?)"
    var statement = conn.prepareStatement(sql)
    statement.set<Int32>(0, 1)
    statement.set<String>(1, "data 1")
    var insert_result = statement.update()
    statement.close()
}

//执行预编译语句，更新数据
func ExecPreparedSQL(conn: Connection) {
    var sql = "UPDATE customer_t1 SET c_customer_name = ? WHERE c_customer_sk = 1"
    var statement = conn.prepareStatement(sql)
    statement.set<String>(0, "new Data")
    var update_result = statement.update()
    statement.close()
}

//查询数据
func QueryTable(conn: Connection) {
    var sql = "select * from customer_t1"
    var statement = conn.prepareStatement(sql)
    let queryResult = statement.query()
    while (queryResult.next()) {
        var c_customer_sk = queryResult.get<Int32>(0)
        var c_customer_name = queryResult.get<String>(1)
        println("c_customer_sk:${c_customer_sk} c_customer_name:${c_customer_name}")
    }
    statement.close()
}

//执行普通SQL语句，删除customer_t1表
func DropTable(conn: Connection) {
    var sql = "DROP TABLE customer_t1"
    var statement = conn.prepareStatement(sql)
    var updateresult = statement.update()
    statement.close()
}

main(): Int64 {
    let conn = GetConnection()

    //创建表。
    CreateTable(conn)

    //批插数据。
    BatchInsertData(conn)

    //执行预编译语句，更新数据。
    ExecPreparedSQL(conn)

    //查询数据
    QueryTable(conn)

    //删除表
    DropTable(conn)

    //关闭数据库连接。
    conn.close()

    return 0
}

```
