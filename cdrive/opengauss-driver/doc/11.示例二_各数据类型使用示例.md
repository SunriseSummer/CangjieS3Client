### 示例二: 各数据类型使用示例

```
import std.database.sql.*
import std.time.*
import std.io.*
import cangjie_tpc::opengauss.driver.*
import cangjie_tpc::opengauss.utils.*


//创建数据库连接
func GetConnection(): Connection {
    var url = "opengauss://user:password@localhost:5432/bes?sslmode=disable&search_path=public"
    let drvopt = DriverManager.getDriver("opengauss")
    var drv: Driver
    match (drvopt) {
        case Some(v) => drv = v
        case None =>
            println("cannot get the opengauss driver")
            throw SqlException("cannot get the opengauss driver")
    }
    let ds = drv.open(url)
    return ds.connect()
}

//执行普通SQL语句，创建all_type_test表
func CreateTable(conn: Connection) {
    var sql = """
        CREATE TABLE IF NOT EXISTS all_type_test
        (
            char1 char(255),
            varchar1 varchar(255),
            clob1 clob,
            blob1 blob,
            boolean1 boolean,
            tinyint1 tinyint,
            smallint1 smallint,
            integer1 integer,
            bigint1 bigint,
            real1 real,
            double1 double precision,
            date1 date,
            time1 time,
            timetz1 timetz,
            timestamp1 timestamp,
            interval1 interval
        );
    """

    var statement = conn.prepareStatement(sql)
    var updateresult = statement.update()
    statement.close()
}

//执行预处理语句，插入数据
func BatchInsertData(conn: Connection) {
    var sql = "insert into all_type_test(char1, varchar1, clob1, blob1, boolean1, tinyint1, smallint1, integer1, bigint1, real1, double1, date1, time1, timetz1, timestamp1, interval1) values(?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)"
    var statement = conn.prepareStatement(sql)

    var char1: String = "09:18:23"
    var varchar1: String = "19:28:33.312"
    var clob1value = "21:58:53.651712"
    var clob1bas = ByteBuffer()
    clob1bas.write(clob1value.toArray())
    var clob1 = clob1bas
    var blob1value = "22:31:41.31242"
    var blob1bas = ByteBuffer()
    blob1bas.write(blob1value.toArray())
    var blob1 = blob1bas
    var boolean1: Bool = true
    var tinyint1: Int8 = 113i8
    var smallint1: Int16 = -21i16
    var integer1: Int32 = -19897i32
    var bigint1: Int64 = 108917i64
    var real1: Float32 = 909.91f32
    var double1: Float64 = 9571781818.9819f64
    var date1tmp: DateTime = DateTime.parse("2020-12-02", "yyyy-MM-dd")
    var date1 = date1tmp
    var time1tmp: DateTime = DateTime.parse("2018-03-09 10:11:51 123456389", "yyyy-MM-dd HH:mm:ss SSS")
    var time1 = time1tmp
    var time1tztmp = DateTime.parse("2016-11-21 21:01:01 223456389 +08", "yyyy-MM-dd HH:mm:ss SSS O")
    var timetz1 = time1tztmp
    var timestamp1tmp = DateTime.parse("2008-11-15 11:20:11 323456389 +08", "yyyy-MM-dd HH:mm:ss SSS O")
    var timestamp1: DateTime = timestamp1tmp
    var interval1: Duration = Duration.second * 28237

    statement.set<String>(0, char1)
    statement.set<String>(1, varchar1)
    statement.set<InputStream>(2, clob1)
    statement.set<InputStream>(3, blob1)
    statement.set<Bool>(4, boolean1)
    statement.set<Int8>(5, tinyint1)
    statement.set<Int16>(6, smallint1)
    statement.set<Int32>(7, integer1)
    statement.set<Int64>(8, bigint1)
    statement.set<Float32>(9, real1)
    statement.set<Float64>(10, double1)
    statement.set<DateTime>(11, date1)
    statement.set<DateTime>(12, time1)
    statement.set<DateTime>(13, timetz1)
    statement.set<DateTime>(14, timestamp1)
    statement.set<Duration>(15, interval1)

    var insert_result = statement.update()
    statement.close()
}

//查询数据
func QueryTable(conn: Connection) {
    var sql = "select char1, varchar1, clob1, blob1, boolean1, tinyint1, smallint1, integer1, bigint1, real1, double1, date1, time1, timetz1, timestamp1, interval1 from all_type_test where varchar1 = ?"
    var statement = conn.prepareStatement(sql)

    var varchar1: String = "19:28:33.312"
    statement.set<String>(0, varchar1)

    let queryResult = statement.query()
    while (queryResult.next()) {
        var char1Got = queryResult.get<String>(0)
        var varchar1Got = queryResult.get<String>(1)
        var clob1Got = queryResult.get<InputStream>(2)
        var blob1Got = queryResult.get<InputStream>(3)
        var boolean1Got = queryResult.get<Bool>(4)
        var tinyint1Got = queryResult.get<Int8>(5)
        var smallint1Got = queryResult.get<Int16>(6)
        var integer1Got = queryResult.get<Int32>(7)
        var bigint1Got = queryResult.get<Int64>(8)
        var real1Got = queryResult.get<Float32>(9)
        var double1Got = queryResult.get<Float64>(10)
        var date1Got = queryResult.get<DateTime>(11)
        var time1Got = queryResult.get<DateTime>(12)
        var timetz1Got = queryResult.get<DateTime>(13)
        var timestamp1Got = queryResult.get<DateTime>(14)
        var interval1Got = queryResult.get<Duration>(15)

        println("${char1Got.toString().trimAscii()}");
        println("${varchar1Got.toString()}");
        println("${inputStreamToStringMax1024(clob1Got)}");
        println("${inputStreamToStringMax1024(blob1Got)}");
        println("${boolean1Got}");
        println("${tinyint1Got.toString()}");
        println("${smallint1Got.toString()}");
        println("${integer1Got}");
        println("${bigint1Got}");
        println("${real1Got.toString()}");
        println("${double1Got.toString()}");
        println("${date1Got.toString()}");
        println("${time1Got.toString()}");
        println("${timetz1Got.toString()}");
        println("${timestamp1Got.toString()}");
        println("zoneId=${timestamp1Got.zoneId}")
        println("${interval1Got.toString()}"); 
    }
    statement.close()
}

//执行普通SQL语句，删除all_type_test表
func DropTable(conn: Connection) {
    var sql = "DROP TABLE all_type_test"
    var statement = conn.prepareStatement(sql)
    var updateresult = statement.update()
    statement.close()
}

main(): Int64 {
    let conn = GetConnection()

    //创建表。
    CreateTable(conn)

    //批插数据。
    BatchInsertData(conn)

    //查询数据
    QueryTable(conn)

    //删除表
    DropTable(conn)

    //关闭数据库连接。
    conn.close()

    return 0
}

func inputStreamToStringMax1024(inputStream: InputStream): String {
    var buf = Array<Byte>(1024, repeat: 0)
    var readlen = inputStream.read(buf)
    return bytesToString(buf, 0, readlen)
}
```
