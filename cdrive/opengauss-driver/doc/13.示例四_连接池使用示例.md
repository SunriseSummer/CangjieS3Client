### 示例四：连接池使用示例

1. cjpm.toml文件内容

```
[package]
  cjc-version = "1.0.0"
  name = "TestPool"
  description = "nothing here"
  version = "1.0.0"
  target-dir = ""
  src-dir = ""
  output-type = "executable"
  compile-option = "-lcrypto --dy-std"
  override-compile-option = ""
  link-option = ""
  package-configuration = {}

[dependencies]
  opengauss = {git = "https://gitcode.com/Cangjie-TPC/opengauss-driver.git", branch = "master"}
```

2. 示例内容

```
package TestPool

import std.database.sql.*
import cangjie_tpc::opengauss.driver.*

main(): Int64 {
    var url = "opengauss://user:password@localhost:5432/bes?sslmode=disable&search_path=public"
    let drvopt = DriverManager.getDriver("opengauss")
    var drv: Driver
    match (drvopt) {
        case Some(v) => drv = v
        case None =>
            println("cannot get the opengauss driver")
            throw SqlException("cannot get the opengauss driver")
    }
    let ds = drv.open(url)

    let pool = PooledDatasource(ds)
    pool.connectionTimeout = Duration.second * 10
    pool.keepaliveTime = Duration.second * 30
    pool.idleTimeout = Duration.second * 60 * 20
    pool.maxLifeTime = Duration.second * 60 * 30
    pool.maxSize = 50
    pool.maxIdleSize = 20

    let conn = pool.connect()
    var sql = "select count(*) from simpletest1"
    var statement = conn.prepareStatement(sql)
    let queryResult = statement.query()
    if (queryResult.next()) {
        let count = queryResult.get<Int32>(0)
        println("count: ${count}")
    }
    statement.close()
    pool.close()

    return 0
}
```
