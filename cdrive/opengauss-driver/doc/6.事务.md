## 事务

### 1 相关原型

```
func createTransaction(): Transaction

public interface Transaction {
    func begin(): Unit                             # 开始数据库事务
    func commit(): Unit                            # 提交数据库事务
    func release(savePointName: String): Unit      # 销毁先前在当前事务中定义的保存点
    func rollback(): Unit                          # 从挂起状态回滚事务
    func rollback(savePointName: String): Unit     # 回滚事务至指定保存点名称
    func save(savePointName: String): Unit         # 在事务中创建一个指定名称的保存点，可用于回滚此保存点之后的事务
}
```



### 2 一个基础的事务示例

```
var transaction = conn.createTransaction()
transaction.begin()

// 插入第一条数据
var sql = "insert into simpletest1(id, updatetime, text1) values(?, ?, ?)"
var statement = conn.prepareStatement(sql)
var id = "testTransaction ${DateTime.now()}"
var updatetime = DateTime.now()
var text1 = "${id + " text1"}"

statement.set<String>(1, id)
statement.set<DateTime>(2, updatetime)
statement.set<String>(3, text1)
var updateresult = statement.update()
statement.close()

// 事务commit操作
transaction.commit()
```



### 3 包含保存点的事务示例

```
var transaction = conn.createTransaction()
transaction.begin()
// 插入第一条数据
var sql = "insert into simpletest1(id, updatetime, text1) values(?, ?, ?)"
var statement = conn.prepareStatement(sql)
var id = "testTransaction ${DateTime.now()}"
var updatetime = DateTime.now()
var text1 = "${id + " text1"}"

statement.set<String>(1, id)
statement.set<DateTime>(2, updatetime)
statement.set<String>(3, text1)
var updateresult = statement.update()

// 保存点
var savePointName = "savepoint1"
transaction.save(savePointName)

// 插入第二条数据
var id2 = "testTransaction2 ${DateTime.now()}"
var updatetime2 = DateTime.now()
var text2 = "${id2 + " text2"}"

statement.set<String>(1, id2)
statement.set<DateTime>(2, updatetime2)
statement.set<String>(3, text2)
updateresult = statement.update()
statement.close()

// 回滚到保存点1
transaction.rollback(savePointName)

// 事务commit操作
transaction.commit()
```
